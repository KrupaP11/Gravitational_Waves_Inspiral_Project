\section{Background Information}
\begin{frame}{Motivation}
  \begin{itemize}
    \item Gravitational-wave (GW) astronomy relies on recovering source parameters from noisy detector data
    \vspace{0.2cm}
    \item Classical parameter estimation (e.g.\ LIGO's matched filtering + Bayesian inference) is \textbf{computationally expensive}
    \vspace{0.2cm}
    \item \textbf{Goal:} Build an ML surrogate model that maps GW inspiral features to source parameters
    \vspace{0.2cm}
  \end{itemize}
\end{frame}

\begin{frame}{Quadrupole Approximation}
  \begin{columns}
    \begin{column}{0.58\textwidth}
      \begin{itemize}
        \item A \textbf{quadrupole moment} describes the asymmetric shape of a mass distribution
        \vspace{0.2cm}
        \item This changing configuration \textbf{radiates gravitational waves} \parencite{Creighton2026}
      \end{itemize}
      \vspace{0.4cm}
      \begin{block}{Key Assumptions}
        \begin{itemize}
          \item Weak-field regime
          \item Non-relativistic velocities ($v \ll c$)
          \item System size $\ll$ gravitational wavelength
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.28\textwidth}
      \hfill
      \begin{tikzpicture}[scale=0.85]
        % Simple dumbbell cartoon
        \draw[darkblue, very thick] (-1.2,0) -- (1.2,0);

      \end{flushright}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Physical Parameters}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{System Parameters}
      \begin{itemize}
        \item Masses: $m_1, m_2 \in [1.2,\,50]\,M_\odot$ 
        \vspace{0.2cm}
        \item Initial Separation: $r_0 \in [5\times10^6,\,10^8]$ km
        \vspace{0.2cm}
        \item Distance to observer: $d \in [10,\,10^5]$ kpc
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Physical Constants}
      \begin{itemize}
        \item $G = 6.674\times10^{-11}$ m$^3$ kg$^{-1}$ s$^{-2}$
        \item $c = 3\times10^8$ m/s
        \item $M_\odot = 1.989\times10^{30}$ kg
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.8cm}
  \begin{itemize}
    \item Circular orbits assumed ($e \approx 0$), consistent with compact binaries \parencite{Peters1964}
    \vspace{0.2cm}
    \item \textbf{1,000 synthetic samples} generated in total
  \end{itemize}
\end{frame}

\begin{frame}{Orbital Decay \& Frequency}
  \begin{columns}
    \begin{column}{0.5\textwidth}
    \vspace{0.2cm}
      \textbf{Orbital Decay} \parencite{Peters1964}
      \[
        \frac{dr}{dt} = -\frac{64}{5}\frac{G^3 m_1 m_2 (m_1+m_2)}{c^5 r^3}
      \]
      \begin{itemize}
        \item Solved numerically via \texttt{scipy} RK45
        \vspace{0.2cm}
        \item Tracks how separation shrinks with time
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Orbital Frequency} (derived in the report)
        \vspace{0.2cm}
      \[
        f_\text{orb} = \frac{1}{2\pi}\sqrt{\frac{G(m_1+m_2)}{r^3}}
      \]
      \begin{itemize}
        \item Derived from $F_g = F_\text{cen}$
        \vspace{0.2cm}
        \item Increases as separation decreases (``chirp'')
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Strain Amplitude \& Target Labels}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{GW Strain} \parencite{Creighton2026}
      \vspace{0.2cm}
      \[
        h(t) = \frac{4(G\mathcal{M})^{5/3}\,\omega_r^{2/3}}{c^4\,d}
      \]
      \begin{itemize}
        \item $\mathcal{M}$: chirp mass
        \item $\omega_r = 2\pi f_\text{orb}$: orbital angular frequency
        \item $d$: distance to observer
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Target Labels}\\[0.5em]
      \vspace{0.2cm}
      Chirp mass:
      \[
        \mathcal{M} = \frac{(m_1 m_2)^{3/5}}{(m_1+m_2)^{1/5}}
      \]
      \vspace{0.2cm}
      Merger time:
      \[
        t_m = \frac{5}{256}\frac{c^5 r_0^4}{G^3 m_1 m_2(m_1+m_2)}
      \]
    \end{column}
  \end{columns}
\end{frame}

\section{Random Forest Model}
% =============================================================================

\begin{frame}{Why Random Forest?}
  \begin{itemize}
    \item \textbf{Ensemble method:} combines many decision trees for better generalization \parencite{Kavlakoglu2025}
    \vspace{0.3cm}
    \item \textbf{Non-linearity:} strain $\to$ chirp mass / merger time mapping is highly non-linear
    \vspace{0.3cm}
    \item \textbf{Scale-invariance:} features span vastly different physical scales and RF handles this naturally
    \vspace{0.3cm}
    \item \textbf{Overfitting mitigation:} ensemble averaging reduces variance with only 1,000 samples
    \vspace{0.3cm}
    \item Implementation: \texttt{scikit-learn} \texttt{RandomForestRegressor}
  \end{itemize}
\end{frame}

\section{Conclusion \& Different Approach}
% =============================================================================

\begin{frame}{Conclusion}
  \begin{block}{Key Findings}
    \begin{itemize}
      \item A Random Forest with \textbf{full features} ($R^2 = 0.87$ train) was exploiting algebraic shortcuts, not learning from strain
      \item Restricting to \textbf{strain only} caused complete model failure ($R^2 \approx 0$)
      \item Real GW parameter estimation uses \textbf{matched filtering} + Bayesian inference (LALInference \parencite{Veitch2015}, Bilby \parencite{Ashton2019}) — far beyond a simple regressor
    \end{itemize}
  \end{block}
  \vspace{0.4em}
  \begin{block}{Lessons Learned}
    \begin{itemize}
      \item Including analytically-derived features \emph{leaks} the answer to the model
      \item 17–20 noisy time samples are \textbf{too sparse} to capture waveform morphology
      \item Feature importance is a vital diagnostic tool
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Different Approach}
  \begin{itemize}
    \item Generate a \textbf{denser strain time-series} (hundreds of time steps spanning the full inspiral)
    \vspace{0.4em}
    \item \textbf{Normalise strain by distance} to remove $d$ as an implicit shortcut
    \vspace{0.4em}
    \item Explore \textbf{other model architectures} that can extract waveform morphology
    \vspace{0.4em}
    \item Compare against LIGO's \textbf{matched filtering} pipeline as a baseline
    \vspace{0.4em}
    \item Use \textbf{Bayesian inference} frameworks (Bilby) to produce posterior distributions over parameters
  \end{itemize}
\end{frame}
